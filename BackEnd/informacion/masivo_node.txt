const { Pool } = require('pg');
const fs = require('fs');
const copyFrom = require('pg-copy-streams').from;

const pool = new Pool({
  user: 'usuario',
  host: 'localhost',
  database: 'tu_basedatos',
  password: 'contraseÃ±a',
  port: 5432,
});

pool.connect((err, client, done) => {
  if (err) throw err;

  const stream = client.query(copyFrom('COPY mi_tabla FROM STDIN WITH CSV HEADER'));
  const fileStream = fs.createReadStream('C:/ruta/archivo.csv');

  fileStream.on('error', err => {
    console.error('Error en el archivo', err);
    done();
  });

  stream.on('error', err => {
    console.error('Error en COPY', err);
    done();
  });

  stream.on('end', () => {
    console.log('Carga masiva completa');
    done();
  });

  fileStream.pipe(stream);
});
